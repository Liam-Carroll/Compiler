program Hanoi;
 var height : integer;
procedure move( height, start, goal, extra : integer); 
begin
if(height > 0)
begin
move (height-1, start, extra, goal);
write(start);
write('to');
write(goal);
write('\n');
call move(height-1, extra, goal, start)
end
end;
begin 
write('How many disks? ');
read(height);
call move(height,1,3,2);
write('\n')
end.
